package net.sf.guavaeclipse.test;

import java.util.Arrays;

public class ExtendedSimpleClass extends SampleSimple {


	private Object objectValue;

	@Override
	public int hashCode() {
		return Arrays
				.deepHashCode(new Object[] { super.hashCode(), objectValue });
	}

	@Override
	public boolean equals(Object object) {
		if (object instanceof ExtendedSimpleClass) {
			if (!super.equals(object))
				return false;
			ExtendedSimpleClass that = (ExtendedSimpleClass) object;
			return Arrays.deepEquals(new Object[] { this.objectValue },
					new Object[] { that.objectValue });
		}
		return false;
	}

}
